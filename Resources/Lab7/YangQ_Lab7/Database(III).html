   Private Sub FirstNameLinkLabel_LinkClicked( ... ) _
            Handles FirstNameLinkLabel.LinkClicked
      Try
         ' Change the color of the link text by setting LinkVisited to True.
         FirstNameLinkLabel.LinkVisited = True
         
         If FirstNameLinkLabel.Text = "Qi" Then
            ' Call the Process.Start method to open the default browser with a URL:
            System.Diagnostics.Process.Start("http://www.uwplatt.edu/~yangq")
         Else
            System.Diagnostics.Process.Start("http://www.uwplatt.edu/csse")
         End If
      Catch ex As Exception
         'MsgBox(ex.Message)
      End Try
   End Sub
